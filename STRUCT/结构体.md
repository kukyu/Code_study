# 结构体

## 定义

结构体的声明

```C

//描述一个学生 需要性别 年龄 电话 性别
// struct 结构体关键字； Stu 结构体标签；struct Stu 结构体类型
struct Stu
{   // 成员变量
    char name[20];
    short age;
    char tele[12];
    char sex[5];
}s1,s2 ,s3; //s1,s2,s3 是三个全局变量（不建议）
int main
{
    struct Stu s; //创建结构体变量 局部变量

    return 0;
}
```

对结构体进行重命名

```C
# include <stdio.h>
//描述一个学生 需要性别 年龄 电话 性别
// struct 结构体关键字； Stu 结构体标签；struct Stu 结构体类型
// 对类型进行重命名
typedef struct Stu
{ // 成员变量
    char name[20];
    short age;
    char tele[12];
    char sex[5];
} Stu; //重新取名叫Stu Stu是类型
int main()
{
    Stu s; //创建结构体变量 局部变量
    return 0;
}
```

## 结构体自引用

```C
# include <stdio.h>
// 结构体的嵌套引用
struct S
{ // 成员变量
    int a;
    char c;
    char arr[20];
    double d;

} ;
struct T
{
    char ch[10];
    struct S s;
    char *p;

};
int main()
{
    char arr[]="hello bit\n";
    struct T t = {"heheh", {100, 'w', "hello world", 3.14}, arr};//初始化
    printf("%s\n", t.ch);
    printf("%s\n",t.s.arr);
    return 0;
}
```

## 结构体传参--函数和函数

```C
#include <stdio.h>
typedef struct Stu
{ // 成员变量
    char name[20];
    short age;
    char tele[12];
    char sex[5];
} Stu; //重新取名叫Stu Stu是类型

void print1(Stu temp) //函数
{
    printf("name:%s\n", temp.name);
    printf("age:%d\n", temp.age);
    printf("Tel:%s\n", temp.tele);
}
void print2(Stu *temp) //指针
{
    printf("name:%s\n", temp->name);//->针对结构体指针
    printf("age:%d\n", temp->age);
    printf("Tel:%s\n", temp->tele);
}
int main()
{
    Stu s = {"张三", 16, "12344555", "男"};
    print1(s);
    print2(&s);
    return 0;
}
```

## 结构体内存对齐

vs 中默认对齐数是 8

结构体大小是最大对齐数的正数倍

拿空间换取时间

`#pragma pack(4) //修改默认对齐数`

`#pragma pack() //取消设置默认对齐数`

```C
#include <stdio.h>
struct S1
{
char c1; // min（1/8） 对齐数是1 （对齐地址是1的倍数）
int a; // 4/8 4  对齐数是4 （对齐地址是4的倍数）
char c2; // 1/8 1 对齐数是1
};

struct S2
{
char c1;
char c2;
int a;
};

int main()
{
struct S1 s1 = {0};
printf("%d\n", sizeof(s1)); // ans=12
struct S2 s2 = {0};
printf("%d\n", sizeof(s2)); // ans=8

    return 0;

}
```
